.MODEL SMALL
.STACK 100H
.DATA
NL EQU 0DH,0AH

ST1 DB "ENTER NUMBERS: $"
ST2 DB NL, "UNIQUE $"
ST3 DB NL, "NOT UNIQUE $"
ARRAY DB ?
.CODE
MAIN PROC
MOV AX,@DATA
MOV DS,AX
;CODE
MOV AH, 9
LEA DX, ST1
INT 21H

MOV CX,5
MOV SI,0
MOV AH,1

INPUT:
INT 21H
MOV BL,AL

MOV ARRAY[SI],BL
INC SI
LOOP INPUT

XOR SI,SI
XOR BX,BX

MOV CX,5


SAVE:
MOV BL,ARRAY[SI]
MOV BH,0
PUSH BX
ADD SI,1
LOOP SAVE


MOV SI,3
MOV CX,5
XOR BX,BX

CHECK:
POP BX
MOV DL,ARRAY[SI]
MOV DH,0
CMP BX,DX
JE PRINTNQ
DEC SI
LOOP CHECK
JMP P
   
   
   
PRINTNQ:
MOV AH, 9
LEA DX, ST3
INT 21H
JMP EXIT


P:
MOV AH, 9
LEA DX, ST2
INT 21H

EXIT:
   
    MAIN ENDP
END MAIN
